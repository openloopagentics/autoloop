# Per-User API Keys & Write Authorization (Sub-project B) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the shared `AUTOLOOP_WRITE_KEYS` with per-user API keys: users mint keys (`/v1/keys`, Firebase ID-token auth), and agent writes are authorized by resolving `API key → user → team membership`.

**Architecture:** Two auth modes under `/v1`. New `requireUser` middleware (verifies a Firebase ID token via the Admin SDK, with an injectable verifier for tests) guards `/v1/keys`. New `requireApiKeyMember` middleware (hash the key → `get apiKeys/{hash}` → check `teams/{teamId}/members/{uid}`) guards the team write subtree, replacing the blanket shared-key guard. Keys are stored as `apiKeys/{sha256(plaintext)}` holding only `{uid,label,prefix,createdAt}`.

**Tech Stack:** TypeScript, Firebase Cloud Functions v2, Express 4, Zod, Firebase Admin SDK (`firebase-admin/auth` + `firebase-admin/firestore`), Vitest + Supertest, Firestore emulator, `@firebase/rules-unit-testing`.

**Reference spec:** `docs/superpowers/specs/2026-06-02-api-keys-design.md`
**Builds on:** the multi-tenant API from `docs/superpowers/specs/2026-06-01-multitenant-foundation-design.md` (already implemented & deployed).

emulator for the full suite; `npm run test:rules` self-launches for rules. For the
inner loop, start `npm run emulators` in the background on port 8080 first.

**Important behavioral shift (drives the test migration in Task 7):** once writes
require membership, **writing to a team the caller isn't a member of returns 403**,
including a team that doesn't exist (you can't be a member of a nonexistent team).
So the existing "404 when the team does not exist" tests become **403**. The
service's own team-existence 404 still fires for a *member* whose project/phase is
missing (those tests stay 404).

```typescript
import { randomBytes, createHash } from "node:crypto";

export const KEY_PREFIX = "al_";

/** A fresh API key: "al_" + 32 random bytes (base64url). Plaintext, shown once. */
export function generateKey(): string {
  return KEY_PREFIX + randomBytes(32).toString("base64url");
}

/** SHA-256 (hex) of the FULL plaintext including the al_ prefix — used as the doc id. */
export function hashKey(plaintext: string): string {
  return createHash("sha256").update(plaintext).digest("hex");
}

/** First 8 chars of the plaintext, stored for display (e.g. "al_ab12c"). */
export function keyDisplayPrefix(plaintext: string): string {
  return plaintext.slice(0, 8);
}
```

```typescript
import { FieldValue } from "firebase-admin/firestore";
import { db } from "../firestore.js";
import { AppError } from "../errors.js";
import { generateKey, hashKey, keyDisplayPrefix } from "../apiKeys.js";

export interface KeySummary {
  id: string;
  label: string;
  prefix: string;
  createdAt: FirebaseFirestore.Timestamp;
}

export async function mintKey(uid: string, label: string): Promise<KeySummary & { key: string }> {
  const key = generateKey();
  const id = hashKey(key);
  const ref = db().doc(`apiKeys/${id}`);
  await ref.set({ uid, label, prefix: keyDisplayPrefix(key), createdAt: FieldValue.serverTimestamp() });
  const data = (await ref.get()).data()!;
  return { id, key, label, prefix: data.prefix, createdAt: data.createdAt };
}

export async function listKeys(uid: string): Promise<KeySummary[]> {
  const q = await db().collection("apiKeys").where("uid", "==", uid).get();
  return q.docs.map((d) => {
    const x = d.data();
    return { id: d.id, label: x.label, prefix: x.prefix, createdAt: x.createdAt };
  });
}

export async function revokeKey(uid: string, id: string): Promise<void> {
  const ref = db().doc(`apiKeys/${id}`);
  const snap = await ref.get();
  if (!snap.exists || snap.data()!.uid !== uid) {
    throw new AppError(404, "not_found", "key not found");
  }
  await ref.delete();
}
```

```typescript
import type { RequestHandler } from "express";
import { getAuth } from "firebase-admin/auth";
import { db } from "./firestore.js";
import { AppError } from "./errors.js";

export type TokenVerifier = (idToken: string) => Promise<{ uid: string }>;

const defaultVerifier: TokenVerifier = (idToken) => getAuth().verifyIdToken(idToken);

/** Verifies a Firebase ID token, requires users/{uid}.isAllowed, sets req.uid. */
export function makeRequireUser(verify: TokenVerifier = defaultVerifier): RequestHandler {
  return async (req, _res, next) => {
    try {
      const auth = req.headers["authorization"];
      const token = typeof auth === "string" && auth.startsWith("Bearer ")
        ? auth.slice("Bearer ".length).trim()
        : undefined;
      if (!token) throw new AppError(401, "unauthorized", "missing ID token");

      let uid: string;
      try {
        ({ uid } = await verify(token));
      } catch {
        throw new AppError(401, "unauthorized", "invalid ID token");
      }

      const snap = await db().doc(`users/${uid}`).get();
      if (!snap.exists || snap.data()?.isAllowed !== true) {
        throw new AppError(403, "forbidden", "user is not allowed");
      }
      req.uid = uid;
      next();
    } catch (err) {
      next(err);
    }
  };
}
```

```typescript
import type { RequestHandler } from "express";
import { db } from "./firestore.js";
import { AppError } from "./errors.js";
import { extractKey } from "./auth.js";
import { hashKey } from "./apiKeys.js";

/** Resolves an API key to its user and authorizes against the path's teamId. */
export const requireApiKeyMember: RequestHandler = async (req, _res, next) => {
  try {
    const key = extractKey(req.headers as Record<string, string | string[] | undefined>);
    if (!key) throw new AppError(401, "unauthorized", "missing API key");

    const keySnap = await db().doc(`apiKeys/${hashKey(key)}`).get();
    if (!keySnap.exists) throw new AppError(401, "unauthorized", "invalid API key");
    const uid = keySnap.data()!.uid as string;

    const { teamId } = req.params as { teamId: string };
    const memberSnap = await db().doc(`teams/${teamId}/members/${uid}`).get();
    if (!memberSnap.exists) throw new AppError(403, "forbidden", "not a member of this team");

    req.uid = uid;
    next();
  } catch (err) {
    next(err);
  }
};
```

```typescript
type Headers = Record<string, string | string[] | undefined>;

export function extractKey(headers: Headers): string | undefined {
  const auth = headers["authorization"];
  if (typeof auth === "string" && auth.startsWith("Bearer ")) {
    return auth.slice("Bearer ".length).trim();
  }

  if (typeof xKey === "string" && xKey.length > 0) return xKey;
  return undefined;
}
```
